\begin{algorithm}
\caption{Computation of the lcp-table from the suffix array and its inverse.}
\label{AlgorithmComputeLcptable}%
\begin{algorithmic}[1]
\State $\Assign{\ell}{0}$
\For{$\Assign{i}{0}$ \Upto $n-1$}
  \State $\Assign{j}{\InverseSUF[i]}$
  \If{$j>0$}
    \State $\Assign{k}{\SUF[j-1]}$ \Comment{$\SF{k}$ directly precedes $\SF{i}$ in $\SUF$}
    \State $\Assign{\ell}{\max\{\ell-1,0\}}$ \Comment{common prefix of length $\geq\ell-1$}\label{ellreference}
    \While{$\Subchar{S}{k+\ell}$ = $\Subchar{S}{i+\ell}$}
      \State $\,\Assign{\ell}{\ell+1}$
    \EndWhile
    \State $\Assign{\LCP[j]}{\ell}$
  \EndIf
\EndFor
\end{algorithmic}
\end{algorithm}
